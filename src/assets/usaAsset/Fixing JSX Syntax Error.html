import { useState, useEffect } from "react";
import winnerANouncement from "../assets/usaAsset/wingo/win_go_win_bg.png";
import LooseANouncement from "../assets/images/wingolostimage.png";
import { ImCross } from "react-icons/im";
import { FaRegCircle } from "react-icons/fa";

const WingoWinnerAnnoucement = ({ data, onClose }) => {
    const [autoCloseTimer, setAutoCloseTimer] = useState(null);

    const handleAutoCloseClick = () => {
        // If a timer is already running, cancel it
        if (autoCloseTimer) {
            clearTimeout(autoCloseTimer);
            setAutoCloseTimer(null);
            return;
        }

        // Start a new 3-second timer to close the modal
        const timer = setTimeout(() => {
            onClose();
            setAutoCloseTimer(null);
        }, 3000);
        
        setAutoCloseTimer(timer);
    };

    useEffect(() => {
        // Cleanup the timer if the component unmounts
        return () => {
            if (autoCloseTimer) clearTimeout(autoCloseTimer);
        };
    }, [autoCloseTimer]);

    return (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center">
            <div
                className="z-50 relative w-80 bg-no-repeat h-[500px] bg-contain bg-center"
                style={{ backgroundImage: `url(${data.win === 0 ? LooseANouncement : winnerANouncement})` }}
            >
                <div className="absolute inset-0 flex flex-col items-center justify-center text-center p-6">

                        </span>
                        &nbsp;&nbsp;

                            {data?.number}
                        </span>
                        &nbsp;&nbsp;
                        <span className={`${data?.number >= 5 ? "bg-green" : "bg-red"} font-bold text-white p-2 rounded-lg`}>
                            {data?.number >= 5 ? "Big" : "Small"}
                        </span>
                    </p>

                    <p className="text-2xl mt-8 text-red font-bold">
                        {data?.win === 0 ? "Lost" : ` ${data?.win?.toFixed(2)}`}
                    </p>

                    <p className="text-xs flex text-blackLight mt-2">
                        Period: <span className="font-bold">{data?.games_no}</span>

                        <ImCross className="border-4 p-1 border-white rounded-full" size={32} />
                    </button>
                </div>
            </div>
        </div>
    );
};

export default WingoWinnerAnnoucement;
